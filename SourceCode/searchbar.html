  3. JS Code: 
    
const search = () => {
    const searchbox = document.getElementById("search-item").value.toUpperCase();
    const storeitems = document.getElementById("anime-list");
    const anime = document.querySelectorAll(".anime");
    const aname = storeitems.getElementsByTagName("h2");

    for(var i = 0; i < aname.length; i++){
        let match = anime[i].getElementsByTagName('h2')[0];

        if(match){
            let textvalue = match.textContent || match.innerHTML

            if(textvalue.toUpperCase().indexOf(searchbox) > -1){
                anime[i].style.display = "";
            }else{
                anime[i].style.display = "none";
            }
        }
    }
}
